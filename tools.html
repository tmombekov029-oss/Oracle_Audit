Инструменты для аудита

  • Oracle SQL Developer — просмотр и экспорт логов
  • DBMS_AUDIT_MGMT — управление и очистка
  • DBMS_FGA — настройка тонкого аудита
  • Интеграция с SIEM (Splunk, ELK, QRadar)
  • PL/SQL скрипты для автоматизации

Примеры SQL/PLSQL


BEGIN
  DBMS_FGA.DROP_POLICY(
    object_schema => 'HR',
    object_name   => 'EMPLOYEES',
    policy_name   => 'salary_read_control'
  );
END;
/
CREATE TABLE audit_archive AS
SELECT * FROM unified_audit_trail
WHERE event_timestamp < SYSDATE - 90;
